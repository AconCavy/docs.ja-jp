## <a name="example"></a><span data-ttu-id="3bafe-160">例</span><span class="sxs-lookup"><span data-stu-id="3bafe-160">Example</span></span>

    <behavior name="TransactedBatchingBehavior"
              maxBatchSize="100" />
  </endpointBehaviors>
