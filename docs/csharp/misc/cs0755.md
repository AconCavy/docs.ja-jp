## <a name="example"></a><span data-ttu-id="e291e-110">例</span><span class="sxs-lookup"><span data-stu-id="e291e-110">Example</span></span>

        static partial void Part(this C c); //Extension method  
  
        // Typically the implementing declaration is in a separate file.

        }  
    }

        }  
    }
