 <span data-ttu-id="c6738-105">正しくないステートメントがコンパイラによって検出されました。</span><span class="sxs-lookup"><span data-stu-id="c6738-105">The compiler found an invalid statement.</span></span>

   public void Dispose()

      Console.WriteLine("Res1.Dispose()");
