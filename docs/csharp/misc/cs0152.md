 <span data-ttu-id="c94f5-107">次の例では CS0152 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="c94f5-107">The following sample generates CS0152:</span></span>  
  
```csharp  
// CS0152.cs  
namespace MyNamespace  
{  
   public class MyClass  
   {  
      public static void Main()  
      {  
         int i = 0;  
  
         switch (i)  
         {  
            case 1:  
               i++;  
               return;  
  
            case 1:   // CS0152, two case 1 statements  
               i++;  
               return;  
         }  
      }  
   }  
}  
```
