 <span data-ttu-id="38155-109">次の例では CS0419 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="38155-109">The following sample generates CS0419.</span></span>  
  
```csharp  
// cs0419.cs  
// compile with: /doc:x.xml /W:3  
interface I  
{  
   /// text for F(void)  
   void F();  
   /// text for F(int)  
   void F(int i);  
}  
/// text for class MyClass  
public class MyClass  
{  
   /// <see cref="I.F"/>  
   public static void MyMethod(int i)  
   {  
   }  
/* Try this instead:  
   /// <see cref="I.F(int)"/>  
   public static void MyMethod(int i)  
   {  
   }  
*/  
   /// text for Main  
   public static void Main ()  
   {  
   }  
}  
```
