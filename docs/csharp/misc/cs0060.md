# <a name="compiler-error-cs0060"></a><span data-ttu-id="3c535-103">コンパイラ エラー CS0060</span><span class="sxs-lookup"><span data-stu-id="3c535-103">Compiler Error CS0060</span></span>

// public class MyClass

   }
