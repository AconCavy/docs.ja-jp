 <span data-ttu-id="b7a47-107">次の例では CS0245 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="b7a47-107">The following sample generates CS0245:</span></span>  
  
```csharp

class MyClass // : IDisposable  
{  
   /*  
   public void Dispose()  
   {  
      // cleanup code goes here

   void m()  
   {  
      this.Finalize();   // CS0245  
      // this.Dispose();  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
