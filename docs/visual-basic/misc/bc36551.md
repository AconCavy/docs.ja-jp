## <a name="example"></a><span data-ttu-id="4db23-108">例</span><span class="sxs-lookup"><span data-stu-id="4db23-108">Example</span></span>

Namespace StringUtility  
    <Extension()> _  
    Module StringExtensions  
        <Extension()> _  
        Public Sub Print (ByVal str As String)  
            Console.WriteLine(str)  
        End Sub  
    End Module  
End Namespace
